Jenkins Pipeline Configuration
pipeline {
    agent any
    
    tools {
        maven 'Maven-3.9.0'
        jdk 'JDK-17'
    }
    
    environment {
        // Anypoint Platform Credentials
        ANYPOINT_CLIENT_ID = credentials('anypoint-client-id')
        ANYPOINT_CLIENT_SECRET = credentials('anypoint-client-secret')
        CLOUDHUB_REGION = 'us-east-1'
        MULE_VERSION = '4.6.0'
    }
    
    parameters {
        choice(
            name: 'DEPLOYMENT_ENVIRONMENT',
            choices: ['dev', 'staging', 'prod'],
            description: 'Select deployment environment'
        )
        choice(
            name: 'MCP_SERVICE',
            choices: ['all', 'employee-onboarding-mcp', 'asset-allocation-mcp'],
            description: 'Select MCP service to deploy'
        )
    }
    
    stages {
        stage('Build & Test') {
            parallel {
                stage('Build MCP Services') {
                    steps {
                        script {
                            def services = params.MCP_SERVICE == 'all' ? 
                                ['employee-onboarding-mcp', 'asset-allocation-mcp'] : 
                                [params.MCP_SERVICE]
                            
                            services.each { service ->
                                dir("mcp-servers/${service}") {
                                    sh "mvn clean compile test package -DskipTests=false"
                                }
                            }
                        }
                    }
                }
                
                stage('Build React Client') {
                    steps {
                        dir('react-client') {
                            sh 'npm ci && npm run build'
                        }
                    }
                }
            }
        }
        
        stage('Deploy to CloudHub') {
            steps {
                script {
                    sh """
                        mvn deploy -DmuleDeploy \\
                            -Dcloudhub.application.name=\${MCP_SERVICE}-\${DEPLOYMENT_ENVIRONMENT} \\
                            -Dcloudhub.environment=\${DEPLOYMENT_ENVIRONMENT} \\
                            -Danypoint.client_id=\${ANYPOINT_CLIENT_ID}
                    """
                }
            }
        }
    }
}

# Employee Onboarding Agent Fabric - GitLab CI/CD Pipeline
image: docker:27.2.0

variables:
  APPLICATION_NAME: "employee-onboarding-agent-fabric"
  BUILD_VERSION: "${CI_PIPELINE_ID}-${CI_COMMIT_SHORT_SHA}"
  MULE_VERSION: "4.9-java17"
  CLOUDHUB_REGION: "us-east-1"

stages:
  - 🔍 validate
  - 🏗️ build
  - 🧪 test
  - 🔍 analyze
  - 📦 package
  - 🐳 containerize
  - 📤 publish
  - 🚀 deploy
  - 🧪 integration
  - ⚡ performance

build:mcp-services:
  stage: 🏗️ build
  image: openjdk:17-jdk-alpine
  parallel:
    matrix:
      - SERVICE: ["employee-onboarding-mcp", "asset-allocation-mcp", "notification-mcp"]
  script:
    - echo "🏗️ Building MCP Service: ${SERVICE}"
    - cd mcp-servers/${SERVICE}
    - mvn clean compile -DskipTests -Dversion=${BUILD_VERSION}
  artifacts:
    paths:
      - mcp-servers/${SERVICE}/target/

test:unit-mcp:
  stage: 🧪 test
  image: openjdk:17-jdk-alpine
  script:
    - echo "🧪 Running Unit Tests for MCP Services"
    - cd mcp-servers/\${SERVICE}
    - mvn test -Dmaven.test.failure.ignore=false
  artifacts:
    reports:
      junit: mcp-servers/*/target/surefire-reports/TEST-*.xml

deploy:cloudhub:prod:
  stage: 🚀 deploy
  image: openjdk:17-jdk-alpine
  script:
    - echo "🚀 Deploying to CloudHub Production"
    - mvn deploy -DmuleDeploy -Dcloudhub.environment=prod
  only:
    - main

Azure DevOps Pipeline (azure-pipelines.yml)
# Employee Onboarding Agent Fabric - Azure DevOps Pipeline
name: Employee-Onboarding-CI-CD-$(Date:yyyyMMdd)$(Rev:.r)

trigger:
  branches:
    include:
    - main
    - develop
    - feature/*
  paths:
    include:
    - mcp-servers/**
    - react-client/**
    - pom.xml

parameters:
- name: serviceSelection
  displayName: 'Service to Build/Deploy'
  type: string
  default: 'all'
  values:
  - all
  - employee-onboarding-mcp
  - asset-allocation-mcp
  - notification-mcp

variables:
  buildVersion: '$(Build.BuildNumber)-$(Build.SourceVersion)'
  muleVersion: '4.9-java17'
  cloudhubRegion: 'us-east-1'

stages:
- stage: Build
  displayName: '🏗️ Build & Compile'
  jobs:
  - job: BuildServices
    displayName: 'Build MCP Services'
    strategy:
      matrix:
        employee-onboarding-mcp:
          serviceName: 'employee-onboarding-mcp'
        asset-allocation-mcp:
          serviceName: 'asset-allocation-mcp'
    steps:
    - task: JavaToolInstaller@0
      displayName: 'Install Java 17'
      inputs:
        versionSpec: '17'
        jdkArchitectureOption: 'x64'
    
    - task: Maven@3
      displayName: 'Build $(serviceName)'
      inputs:
        mavenPomFile: 'mcp-servers/$(serviceName)/pom.xml'
        goals: 'clean compile test package'
        options: '-B -Dversion=$(buildVersion)'

- stage: Deploy
  displayName: '🚀 Deploy to CloudHub'
  jobs:
  - deployment: CloudHubDeployment
    environment: production
    strategy:
      runOnce:
        deploy:
          steps:
          - task: Maven@3
            displayName: 'Deploy to CloudHub'
            inputs:
              goals: 'deploy -DmuleDeploy'
              options: '-Dcloudhub.environment=prod'

GitHub Actions Workflow (.github/workflows/ci-cd.yml)
name: Employee Onboarding MCP CI/CD

on:
  push:
    branches: [ main, develop ]
    paths:
      - 'mcp-servers/**'
      - 'react-client/**'
      - 'pom.xml'
  pull_request:
    branches: [ main, develop ]
  workflow_dispatch:
    inputs:
      service:
        description: 'Service to deploy'
        required: true
        default: 'all'
        type: choice
        options:
        - all
        - employee-onboarding-mcp
        - asset-allocation-mcp

env:
  MULE_VERSION: '4.9-java17'
  CLOUDHUB_REGION: 'us-east-1'
  BUILD_VERSION: '${{ github.run_number }}-${{ github.sha }}'

jobs:
  build:
    name: Build MCP Services
    runs-on: ubuntu-latest
    strategy:
      matrix:
        service: [employee-onboarding-mcp, asset-allocation-mcp, notification-mcp]
    
    steps:
    - name: Checkout Code
      uses: actions/checkout@v4
    
    - name: Setup Java 17
      uses: actions/setup-java@v4
      with:
        java-version: '17'
        distribution: 'temurin'
    
    - name: Cache Maven Dependencies
      uses: actions/cache@v3
      with:
        path: ~/.m2
        key: ${{ runner.os }}-m2-${{ hashFiles('**/pom.xml') }}
    
    - name: Build ${{ matrix.service }}
      working-directory: mcp-servers/${{ matrix.service }}
      run: |
        mvn clean compile test package \
          -B -Dversion=${{ env.BUILD_VERSION }} \
          -Dmule.version=${{ env.MULE_VERSION }}
    
    - name: Upload Build Artifacts
      uses: actions/upload-artifact@v4
      with:
        name: ${{ matrix.service }}-build
        path: mcp-servers/${{ matrix.service }}/target/

  security-scan:
    name: Security Scanning
    runs-on: ubuntu-latest
    needs: build
    
    steps:
    - name: Checkout Code
      uses: actions/checkout@v4
    
    - name: Run Snyk Security Scan
      uses: snyk/actions/maven@master
      env:
        SNYK_TOKEN: ${{ secrets.SNYK_TOKEN }}
    
    - name: Run OWASP Dependency Check
      run: |
        mvn org.owasp:dependency-check-maven:check \
          -DfailBuildOnCVSS=8

  deploy:
    name: Deploy to CloudHub
    runs-on: ubuntu-latest
    needs: [build, security-scan]
    if: github.ref == 'refs/heads/main'
    environment: production
    
    steps:
    - name: Deploy MCP Services
      run: |
        mvn deploy -DmuleDeploy \
          -Dcloudhub.environment=prod \
          -Danypoint.client_id=${{ secrets.ANYPOINT_CLIENT_ID }} \
          -Danypoint.client_secret=${{ secrets.ANYPOINT_CLIENT_SECRET }}
