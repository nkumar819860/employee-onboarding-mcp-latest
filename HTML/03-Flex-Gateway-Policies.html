🔧 JWT Validation Configuration Example

# Flex Gateway JWT Validation Policy apiVersion: gateway.mulesoft.com/v1alpha1 kind: ApiInstance metadata: name: employee-onboarding-api spec: address: https://api.employee-onboarding.com services: employee-service: address: https://employees.cloudhub.io:443 routes: - rules: - path: /api/employees/* - path: /api/onboarding/* policies: - policyRef: name: jwt-validation-flex config: jwtKeyOrigin: jwks jwksUrl: "https://auth.employee-portal.com/.well-known/jwks.json" jwksServiceTimeToLive: 60 audience: "employee-onboarding-api" issuer: "https://auth.employee-portal.com/" validateAudience: true validateIssuer: true mandatoryClaims: - "sub" - "iat" - "exp" - "employee_role" - policyRef: name: rate-limiting-flex config: rateLimits: - maximumRequests: 1000 timePeriod: 1 timePeriodTimeUnit: MINUTE identifier: "#[attributes.headers['client-id']]" quotas: - maximumRequests: 10000 timePeriod: 1 timePeriodTimeUnit: HOUR identifier: "#[attributes.headers['client-id']]" - policyRef: name: pii-protection-flex config: encryptionKey: "${secure::pii.encryption.key}" protectedFields: - "ssn" - "email" - "phone" - "address" auditEnabled: true hipaaCompliance: true
