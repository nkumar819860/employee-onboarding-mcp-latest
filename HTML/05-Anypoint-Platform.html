🏢 Organization & Business Group Setup

📋 Step 1: Create Business Group for Employee Onboarding

# Anypoint Platform Organization Configuration #!/bin/bash # Step 1: Access Management Setup # Create Business Groups for Employee Onboarding curl -X POST "https://anypoint.mulesoft.com/accounts/api/organizations/${MASTER_ORG_ID}/groups" \ -H "Authorization: Bearer ${ACCESS_TOKEN}" \ -H "Content-Type: application/json" \ -d '{ "name": "Employee Onboarding Division", "parentOrganizationId": "'${MASTER_ORG_ID}'", "ownerId": "'${USER_ID}'", "entitlements": { "runtimeFabric": { "assignedCores": 24, "assignedMemory": "96Gi" }, "apis": { "maxCount": 100 } } }'

👥 Step 2: Configure User Roles and Permissions

# Setup user roles for Employee Onboarding # Admin role for platform management curl -X POST "https://anypoint.mulesoft.com/accounts/api/organizations/${ORG_ID}/rolegroups" \ -H "Authorization: Bearer ${ACCESS_TOKEN}" \ -H "Content-Type: application/json" \ -d '{ "name": "Employee Onboarding Admin", "description": "Full access to employee onboarding platform", "external_names": [], "roles": [ { "role_id": "organization_admin" }, { "role_id": "runtime_fabric_admin" } ] }' # Developer role for application deployment curl -X POST "https://anypoint.mulesoft.com/accounts/api/organizations/${ORG_ID}/rolegroups" \ -H "Authorization: Bearer ${ACCESS_TOKEN}" \ -H "Content-Type: application/json" \ -d '{ "name": "Employee Onboarding Developer", "description": "Development access to employee onboarding platform", "roles": [ { "role_id": "runtime_fabric_developer" }, { "role_id": "exchange_contributor" } ] }'

🌍 Environment Configuration & Setup

🏗️ Multi-Environment Setup

# Step 2: Create Environments with Enhanced Configuration environments=("development" "testing" "staging" "production") environment_types=("SANDBOX" "SANDBOX" "SANDBOX" "PRODUCTION") for i in "${!environments[@]}"; do env="${environments[$i]}" env_type="${environment_types[$i]}" echo "Creating environment: $env" curl -X POST "https://anypoint.mulesoft.com/accounts/api/organizations/${ORG_ID}/environments" \ -H "Authorization: Bearer ${ACCESS_TOKEN}" \ -H "Content-Type: application/json" \ -d '{ "name": "'${env}'", "type": "'${env_type}'", "isProduction": '$([ "$env" = "production" ] && echo "true" || echo "false")', "clientId": "'${CONNECTED_APP_CLIENT_ID}'" }' done

🎯 RTF Target Configuration & Association

🔗 Associate RTF with Environments

# Associate RTF with Environment curl -X POST "https://anypoint.mulesoft.com/hybrid/api/v1/organizations/${ORG_ID}/environments/${ENV_NAME}/deploymentTargets" \ -H "Authorization: Bearer ${ACCESS_TOKEN}" \ -H "Content-Type: application/json" \ -d '{ "name": "RTF Production EKS", "type": "MC", "deploymentSettings": { "runtimeVersion": "4.9.0", "autoScaling": { "enabled": true, "minReplicas": 3, "maxReplicas": 12, "targetCPUUtilization": 80, "targetMemoryUtilization": 80 }, "resources": { "cpu": { "reserved": "1000m", "limit": "2000m" }, "memory": { "reserved": "1Gi", "limit": "4Gi" } }, "networking": { "inbound": { "http": { "port": 8081, "protocol": "HTTP" }, "https": { "port": 8082, "protocol": "HTTPS" } } } } }'

⚙️ Configure RTF Properties

# Configure RTF-specific properties curl -X PUT "https://anypoint.mulesoft.com/hybrid/api/v1/organizations/${ORG_ID}/targets/${RTF_TARGET_ID}" \ -H "Authorization: Bearer ${ACCESS_TOKEN}" \ -H "Content-Type: application/json" \ -d '{ "properties": [ { "key": "anypoint.platform.config.analytics.agent.enabled", "value": "true", "secure": false }, { "key": "anypoint.platform.visualizer.layer", "value": "Employee Onboarding Platform", "secure": false }, { "key": "encryption.key", "value": "${secure::encryption.key}", "secure": true }, { "key": "hipaa.compliance.enabled", "value": "true", "secure": false }, { "key": "auto.scaling.threshold.cpu", "value": "80", "secure": false }, { "key": "auto.scaling.threshold.memory", "value": "80", "secure": false } ] }'

🚀 Application Deployment Workflow

📋 Employee Onboarding Platform Deployment Process: Build Application → Upload to Exchange → Deploy to RTF → Configure Scaling → Validate Deployment

📦 Deploy Employee Onboarding MCP Services

#!/bin/bash # Array of MCP services to deploy services=( "employee-onboarding-mcp:1.0.0" "asset-allocation-mcp:1.0.0" "notification-mcp:1.0.0" "agent-broker-mcp:1.0.0" ) for service in "${services[@]}"; do IFS=':' read -ra ADDR <<< "$service" app_name="${ADDR[0]}" version="${ADDR[1]}" echo "Deploying $app_name version $version..." curl -X POST "https://anypoint.mulesoft.com/hybrid/api/v2/organizations/${ORG_ID}/environments/${ENV_ID}/deployments" \ -H "Authorization: Bearer ${ACCESS_TOKEN}" \ -H "Content-Type: application/json" \ -d '{ "name": "'${app_name}'", "target": { "targetId": "'${RTF_TARGET_ID}'", "deploymentSettings": { "runtimeVersion": "4.9.0", "resources": { "cpu": { "reserved": "500m", "limit": "1000m" }, "memory": { "reserved": "1Gi", "limit": "2Gi" } }, "clustered": false, "enforceDeployingReplicasAcrossNodes": false, "http": { "inbound": { "publicUrl": "https://'${app_name}'-'${ENV_NAME}'.cloudhub.io" } }, "jvm": { "args": "-XX:+UseG1GC -XX:+UseStringDeduplication" }, "properties": { "auto.scaling.enabled": "true", "auto.scaling.min.replicas": "1", "auto.scaling.max.replicas": "3", "auto.scaling.target.cpu": "80", "auto.scaling.target.memory": "80" } } }, "application": { "ref": { "groupId": "'${GROUP_ID}'", "artifactId": "'${app_name}'", "version": "'${version}'" } } }' echo "Deployment request submitted for $app_name" echo "Waiting for deployment to complete..." sleep 30 done

📊 Deployment Status Monitoring

# Monitor deployment status monitor_deployment_status() { local app_name=$1 local max_attempts=20 local attempt=0 while [ $attempt -lt $max_attempts ]; do status=$(curl -s -X GET \ "https://anypoint.mulesoft.com/hybrid/api/v2/organizations/${ORG_ID}/environments/${ENV_ID}/deployments" \ -H "Authorization: Bearer ${ACCESS_TOKEN}" | \ jq -r ".items[] | select(.name==\"$app_name\") | .status") case $status in "STARTED") echo "✅ $app_name: Deployment successful" return 0 ;; "FAILED") echo "❌ $app_name: Deployment failed" return 1 ;; "STARTING") echo "🔄 $app_name: Still starting... (attempt $((attempt+1))/$max_attempts)" ;; *) echo "⏳ $app_name: Status - $status" ;; esac sleep 15 ((attempt++)) done echo "⚠️ $app_name: Deployment timeout after $((max_attempts * 15)) seconds" return 1 } # Monitor all services for service in "${services[@]}"; do IFS=':' read -ra ADDR <<< "$service" app_name="${ADDR[0]}" monitor_deployment_status "$app_name" done
