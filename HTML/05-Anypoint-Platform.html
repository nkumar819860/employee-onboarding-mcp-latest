# Step 2: Create Environments with Enhanced Configuration
environments=("development" "testing" "staging" "production")
environment_types=("SANDBOX" "SANDBOX" "SANDBOX" "PRODUCTION")

for i in "${!environments[@]}"; do
  env="${environments[$i]}"
  env_type="${environment_types[$i]}"
  
  echo "Creating environment: $env"
  curl -X POST "https://anypoint.mulesoft.com/accounts/api/organizations/${ORG_ID}/environments" \
    -H "Authorization: Bearer ${ACCESS_TOKEN}" \
    -H "Content-Type: application/json" \
    -d '{
      "name": "'${env}'",
      "type": "'${env_type}'",
      "isProduction": '$([ "$env" = "production" ] && echo "true" || echo "false")',
      "clientId": "'${CONNECTED_APP_CLIENT_ID}'"
    }'
done

# Deploy Employee Onboarding MCP Services
#!/bin/bash

# Array of MCP services to deploy
services=(
  "employee-onboarding-mcp:1.0.0"
  "asset-allocation-mcp:1.0.0"
  "notification-mcp:1.0.0"
  "agent-broker-mcp:1.0.0"
)

for service in "${services[@]}"; do
  IFS=':' read -ra ADDR <<< "$service"
  app_name="${ADDR[0]}"
  version="${ADDR[1]}"
  
  echo "Deploying $app_name version $version..."
  
  curl -X POST "https://anypoint.mulesoft.com/hybrid/api/v2/organizations/${ORG_ID}/environments/${ENV_ID}/deployments" \
    -H "Authorization: Bearer ${ACCESS_TOKEN}" \
    -H "Content-Type: application/json" \
    -d '{
      "name": "'${app_name}'",
      "target": {
        "targetId": "'${RTF_TARGET_ID}'",
        "deploymentSettings": {
          "runtimeVersion": "4.9.0",
          "resources": {
            "cpu": {
              "reserved": "500m",
              "limit": "1000m"
            },
            "memory": {
              "reserved": "1Gi", 
              "limit": "2Gi"
            }
          },
          "clustered": false,
          "enforceDeployingReplicasAcrossNodes": false,
          "http": {
            "inbound": {
              "publicUrl": "https://'${app_name}'-'${ENV_NAME}'.cloudhub.io"
            }
          },
          "jvm": {
            "args": "-XX:+UseG1GC -XX:+UseStringDeduplication"
          },
          "properties": {
            "auto.scaling.enabled": "true",
            "auto.scaling.min.replicas": "1",
            "auto.scaling.max.replicas": "3", 
            "auto.scaling.target.cpu": "80",
            "auto.scaling.target.memory": "80"
          }
        }
      },
      "application": {
        "ref": {
          "groupId": "'${GROUP_ID}'",
          "artifactId": "'${app_name}'",
          "version": "'${version}'"
        }
      }
    }'
  
  echo "Deployment request submitted for $app_name"
  echo "Waiting for deployment to complete..."
  sleep 30
done
4.2 Deployment Status Monitoring
# Monitor deployment status
monitor_deployment_status() {
  local app_name=$1
  local max_attempts=20
  local attempt=0
  
  while [ $attempt -lt $max_attempts ]; do
    status=$(curl -s -X GET \
      "https://anypoint.mulesoft.com/hybrid/api/v2/organizations/${ORG_ID}/environments/${ENV_ID}/deployments" \
      -H "Authorization: Bearer ${ACCESS_TOKEN}" | \
      jq -r ".items[] | select(.name==\"$app_name\") | .status")
    
    case $status in
      "STARTED")
        echo "✅ $app_name: Deployment successful"
        return 0
        ;;
      "FAILED")
        echo "❌ $app_name: Deployment failed"
        return 1
        ;;
      "STARTING")
        echo "🔄 $app_name: Still starting... (attempt $((attempt+1))/$max_attempts)"
        ;;
      *)
        echo "⏳ $app_name: Status - $status"
        ;;
    esac
    
    sleep 15
    ((attempt++))
  done
  
  echo "⚠️ $app_name: Deployment timeout after $((max_attempts * 15)) seconds"
  return 1
}

# Monitor all services
for service in "${services[@]}"; do
  IFS=':' read -ra ADDR <<< "$service"
  app_name="${ADDR[0]}"
  monitor_deployment_status "$app_name"
done
