Step-by-step guide for setting up Runtime Fabric on Amazon EKS from scratch, including cluster creation, RTF installation, and Anypoint Platform integration with 80% auto-scaling thresholds.

2 🚀 EKS Cluster Creation with RTF Specifications 
# Enhanced EKS Cluster Configuration for RTF
cat < rtf-production-cluster.yaml
apiVersion: eksctl.io/v1alpha5
kind: ClusterConfig

metadata:
  name: rtf-production-cluster
  region: us-east-1
  version: "1.28"

# VPC Configuration
vpc:
  cidr: "10.0.0.0/16"
  nat:
    gateway: HighlyAvailable
  clusterEndpoints:
    publicAccess: true
    privateAccess: true
  publicAccessCIDRs: ["0.0.0.0/0"]

# Enhanced Node Groups for RTF
managedNodeGroups:
  # RTF Controller Nodes (Optimized for Control Plane)
  - name: rtf-controllers
    instanceType: c5.2xlarge  # 8 vCPU, 16GB RAM
    minSize: 3
    maxSize: 3
    desiredCapacity: 3
    volumeSize: 100
    volumeType: gp3
    amiFamily: AmazonLinux2
    labels:
      node-type: controller
      rtf-role: controller
      workload: control-plane
    taints:
      - key: rtf-controller
        value: "true"
        effect: NoSchedule
    tags:
      Environment: production
      Application: runtime-fabric
      Role: controller
    availabilityZones: ["us-east-1a", "us-east-1b", "us-east-1c"]
    
  # RTF Worker Nodes (Optimized for Application Workloads) 
  - name: rtf-workers
    instanceType: m5.2xlarge  # 8 vCPU, 32GB RAM
    minSize: 3
    maxSize: 12  # Auto-scaling capacity
    desiredCapacity: 6
    volumeSize: 200
    volumeType: gp3
    amiFamily: AmazonLinux2
    labels:
      node-type: worker
      rtf-role: worker
      workload: application
    tags:
      Environment: production
      Application: runtime-fabric
      Role: worker
    availabilityZones: ["us-east-1a", "us-east-1b", "us-east-1c"]
    
    # Auto-scaling Configuration (80% threshold)
    maxPodsPerNode: 110
    kubeletExtraConfig:
      kubeReserved:
        cpu: "200m"
        memory: "1Gi"
        ephemeral-storage: "1Gi"
      systemReserved:
        cpu: "200m"
        memory: "1Gi"
        ephemeral-storage: "1Gi"
      evictionHard:
        memory.available: "100Mi"
        nodefs.available: "10%"
EOF

# Create the EKS Cluster
echo "🚀 Creating EKS cluster for RTF..."
eksctl create cluster -f rtf-production-cluster.yaml
⚠️ RTF EKS Best Practices
  • Resource Limits: Set CPU/Memory limits to 80% of node capacity
  • Node Groups: Separate controller and worker nodes with taints
  • Storage: Use GP3 volumes for better performance and cost
  • Networking: Enable private subnets for enhanced security
  • Monitoring: Enable detailed monitoring and logging
3 🎯 RTF Installation & Configuration
3.1 Install RTF Operator
# Download and install RTF operator
curl -L https://runtime-fabric.s3.amazonaws.com/installer/rtf-install-scripts.tar.gz -o rtf-install-scripts.tar.gz
tar -xzf rtf-install-scripts.tar.gz
cd rtf-install-scripts

# Install RTF operator
./rtfctl install --activation-data="YOUR_ACTIVATION_DATA_HERE"

# Verify installation
kubectl get pods -n rtf-system
3.2 Configure RTF with Auto-scaling
# RTF Configuration with 80% Auto-scaling Thresholds
apiVersion: install.rtf.mulesoft.com/v1
kind: RuntimeFabric
metadata:
  name: rtf-production
  namespace: rtf-system
spec:
  # Cluster Configuration
  cluster:
    name: "rtf-production-cluster"
    region: "us-east-1"
    
  # Auto-scaling Configuration (80% threshold)
  autoscaling:
    enabled: true
    minReplicas: 3
    maxReplicas: 12
    targetCPUUtilizationPercentage: 80
    targetMemoryUtilizationPercentage: 80
    scaleUpPolicy:
      stabilizationWindowSeconds: 60
      selectPolicy: Max
      policies:
      - type: Percent
        value: 100
        periodSeconds: 15
    scaleDownPolicy:
      stabilizationWindowSeconds: 300
      selectPolicy: Min
      policies:
      - type: Percent
        value: 20
        periodSeconds: 60

  # Resource Allocation
  resources:
    controllers:
      cpu: "2000m"
      memory: "4Gi"
    workers:
      cpu: "6000m"
      memory: "24Gi"
      
  # Storage Configuration
  storage:
    storageClass: "gp3"
    size: "100Gi"
    
  # Security Configuration
  security:
    rbacEnabled: true
    podSecurityPolicyEnabled: true
3.3 Apply RTF Configuration
# Apply RTF configuration
kubectl apply -f rtf-production-config.yaml

# Wait for RTF to be ready
kubectl wait --for=condition=Ready runtimefabric/rtf-production --timeout=1800s

# Verify RTF status
kubectl get runtimefabric -n rtf-system
kubectl get pods -n rtf-system
4 🔍 Validation & Testing
4.1 Cluster Health Validation
# Comprehensive RTF EKS Validation Script
#!/bin/bash

echo "🔍 RTF EKS Validation Suite"
echo "============================="

# Check EKS cluster health
echo "1. Checking EKS cluster connectivity..."
kubectl cluster-info
if [ $? -eq 0 ]; then
    echo "✅ EKS cluster is accessible"
else
    echo "❌ EKS cluster connection failed"
    exit 1
fi

# Check RTF operator status
echo "2. Checking RTF operator status..."
RTF_PODS=$(kubectl get pods -n rtf-system --no-headers | wc -l)
RTF_READY=$(kubectl get pods -n rtf-system --field-selector=status.phase=Running --no-headers | wc -l)

if [ "$RTF_READY" -eq "$RTF_PODS" ]; then
    echo "✅ RTF operator: $RTF_READY/$RTF_PODS pods ready"
else
    echo "⚠️ RTF operator: $RTF_READY/$RTF_PODS pods ready"
fi

# Check auto-scaling configuration
echo "3. Validating auto-scaling configuration..."
HPA_COUNT=$(kubectl get hpa -n rtf-system --no-headers | wc -l)
if [ "$HPA_COUNT" -gt 0 ]; then
    echo "✅ Auto-scaling configured: $HPA_COUNT HPA found"
    
    # Check 80% threshold
    kubectl get hpa -n rtf-system -o json | jq -r '.items[] | select(.spec.targetCPUUtilizationPercentage == 80)'
    if [ $? -eq 0 ]; then
        echo "✅ 80% CPU threshold configured"
    else
        echo "⚠️ 80% threshold not found"
    fi
else
    echo "❌ No HPA configurations found"
fi

# Check node readiness
echo "4. Checking node readiness..."
NODE_COUNT=$(kubectl get nodes --no-headers | wc -l)
READY_NODES=$(kubectl get nodes --no-headers | grep " Ready " | wc -l)

if [ "$READY_NODES" -eq "$NODE_COUNT" ]; then
    echo "✅ All nodes ready: $READY_NODES/$NODE_COUNT"
else
    echo "⚠️ Nodes ready: $READY_NODES/$NODE_COUNT"
fi

# Check RTF fabric status
echo "5. Checking RTF fabric status..."
kubectl get runtimefabric -n rtf-system
if [ $? -eq 0 ]; then
    echo "✅ RTF fabric status retrieved"
else
    echo "❌ RTF fabric status check failed"
fi

echo "============================="
echo "🎯 RTF EKS validation completed"
4.2 Performance Validation
# Performance and Resource Validation
echo "📊 Performance Validation"

# Check resource usage
echo "CPU and Memory Usage:"
kubectl top nodes

# Check auto-scaling metrics
echo "Auto-scaling Metrics:"
kubectl get hpa -n rtf-system -o wide

# Load test simulation (optional)
echo "Running load test simulation..."
kubectl run load-test --rm -i --restart=Never --image=busybox -- /bin/sh -c "while true; do echo 'load test'; sleep 1; done" &

# Monitor scaling behavior
for i in {1..60}; do
    echo "Minute $i: Checking replica count..."
    kubectl get pods -n rtf-system | grep -c "Running"
    sleep 60
done
