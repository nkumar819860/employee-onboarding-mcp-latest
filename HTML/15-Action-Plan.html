# Phase 1 - Infrastructure as Code (Terraform)
resource "aws_eks_cluster" "mulesoft_rtf" {
  name     = "mulesoft-rtf-cluster"
  role_arn = aws_iam_role.cluster.arn
  version  = "1.28"

  vpc_config {
    subnet_ids              = aws_subnet.private[*].id
    endpoint_private_access = true
    endpoint_public_access  = true
    security_group_ids      = [aws_security_group.cluster.id]
  }

  encryption_config {
    provider {
      key_arn = aws_kms_key.rtf.arn
    }
    resources = ["secrets"]
  }

  enabled_cluster_log_types = [
    "api", "audit", "authenticator", 
    "controllerManager", "scheduler"
  ]
}

# Phase 1 - Node Groups
resource "aws_eks_node_group" "rtf_nodes" {
  cluster_name    = aws_eks_cluster.mulesoft_rtf.name
  node_group_name = "rtf-worker-nodes"
  node_role_arn   = aws_iam_role.node.arn
  subnet_ids      = aws_subnet.private[*].id

  capacity_type  = "ON_DEMAND"
  instance_types = ["m5.2xlarge"]

  scaling_config {
    desired_size = 3
    max_size     = 12
    min_size     = 3
  }

  update_config {
    max_unavailable = 1
  }
}

# Phase 2 - Runtime Fabric Deployment
apiVersion: v1
kind: Namespace
metadata:
  "name": rtf-system
  labels:
    app.kubernetes.io/name: runtime-fabric
    rtf.mulesoft.com/version: "1.15.x"

---
apiVersion: apps/v1
kind: Deployment
metadata:
  "name": runtime-fabric-controller
  namespace: rtf-system
spec:
  replicas: 3
  selector:
    matchLabels:
      app: rtf-controller
  template:
    metadata:
      labels:
        app: rtf-controller
        version: "1.15.x"
    spec:
      serviceAccountName: rtf-controller
      containers:
      - name: controller
        image: mulesoft/runtime-fabric:1.15.x
        resources:
          requests:
            memory: "2Gi"
            cpu: "1"
          limits:
            memory: "4Gi"
            cpu: "2"
        env:
        - name: RTF_ACTIVATION_DATA
          valueFrom:
            secretKeyRef:
              "name": rtf-activation
              key: data
        - name: RTF_NODE_ROLE
          value: "controller_node"

🛠️ Auto-scaling Configuration

# Phase 3 - Advanced HPA Configuration
apiVersion: autoscaling/v2
kind: HorizontalPodAutoscaler
metadata:
  "name": employee-onboarding-hpa-advanced
  namespace: employee-onboarding
  annotations:
    predictive.autoscaling/enabled: "true"
    hipaa.compliance/audit-enabled: "true"
spec:
  scaleTargetRef:
    apiVersion: apps/v1
    kind: Deployment
    "name": employee-onboarding-api
  "minReplicas": 3
  "maxReplicas": 12
  metrics:
  - type: Resource
    resource:
      "name": cpu
      target:
        "type": Utilization
        averageUtilization: 80
  - type: Resource
    resource:
      "name": memory
      target:
        "type": Utilization
        averageUtilization: 80
  - type: Pods
    pods:
      metric:
        "name": employee_onboarding_requests_per_second
      target:
        "type": AverageValue
        averageValue: "100"
  behavior:
    scaleUp:
      stabilizationWindowSeconds: 60
      policies:
      - type: Percent
        value: 100
        periodSeconds: 15
    scaleDown:
      stabilizationWindowSeconds: 300
      policies:
      - type: Percent
        value: 20
        periodSeconds: 60
