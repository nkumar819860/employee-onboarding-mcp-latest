🔍 ELK Stack Deployment

📦 Step 1: Deploy Elasticsearch Cluster

💡 Production Setup: Deploy a 3-node Elasticsearch cluster for high availability and data redundancy.
# Add Elastic Helm repository helm repo add elastic https://helm.elastic.co helm repo update # Create namespace for ELK stack kubectl create namespace logging # Deploy Elasticsearch with production configuration cat << EOF > elasticsearch-values.yaml clusterName: "rtf-elasticsearch" nodeGroup: "master" # Master-eligible nodes master: enabled: true replicas: 3 resources: requests: cpu: "1000m" memory: "2Gi" limits: cpu: "2000m" memory: "4Gi" # Data nodes data: enabled: true replicas: 3 resources: requests: cpu: "1500m" memory: "4Gi" limits: cpu: "3000m" memory: "8Gi" # Storage configuration volumeClaimTemplate: accessModes: [ "ReadWriteOnce" ] storageClassName: "rtf-premium-storage" resources: requests: storage: 100Gi # Security settings esConfig: elasticsearch.yml: | xpack.security.enabled: true xpack.security.transport.ssl.enabled: true xpack.security.transport.ssl.verification_mode: certificate xpack.security.transport.ssl.keystore.path: /usr/share/elasticsearch/config/certs/elastic-keystore.p12 xpack.security.transport.ssl.truststore.path: /usr/share/elasticsearch/config/certs/elastic-keystore.p12 xpack.security.http.ssl.enabled: true xpack.security.http.ssl.keystore.path: /usr/share/elasticsearch/config/certs/elastic-keystore.p12 # JVM settings for optimal performance esJavaOpts: "-Xms2g -Xmx2g" # Service configuration service: type: ClusterIP ports: - port: 9200 name: http - port: 9300 name: transport EOF # Install Elasticsearch helm install elasticsearch elastic/elasticsearch \ --namespace logging \ --values elasticsearch-values.yaml \ --wait \ --timeout 10m # Verify Elasticsearch deployment kubectl get pods -n logging -l app=elasticsearch-master kubectl get svc -n logging elasticsearch-master

🚰 Step 2: Deploy Logstash

# Create Logstash configuration cat << EOF > logstash-config.yaml apiVersion: v1 kind: ConfigMap metadata: name: logstash-config namespace: logging data: logstash.yml: | http.host: "0.0.0.0" xpack.monitoring.elasticsearch.hosts: ["http://elasticsearch-master:9200"] pipelines.yml: | - pipeline.id: rtf-logs path.config: "/usr/share/logstash/pipeline/rtf-logs.conf" - pipeline.id: kubernetes-logs path.config: "/usr/share/logstash/pipeline/k8s-logs.conf" rtf-logs.conf: | input { beats { port => 5044 } tcp { port => 5000 codec => json_lines } } filter { if [kubernetes] { mutate { add_field => { "cluster_name" => "rtf-cluster" } } if [kubernetes][container_name] == "runtime-fabric" { grok { match => { "message" => "%{TIMESTAMP_ISO8601:timestamp} %{LOGLEVEL:level} \[%{DATA:thread}\] %{DATA:logger}: %{GREEDYDATA:message}" } } date { match => [ "timestamp", "ISO8601" ] } } if [kubernetes][container_name] =~ /mule-/ { grok { match => { "message" => "%{TIMESTAMP_ISO8601:timestamp} %{LOGLEVEL:level} \[%{DATA:thread}\] %{DATA:flow}: %{GREEDYDATA:message}" } } } } # Add GeoIP information for external IPs if [remote_addr] and [remote_addr] !~ /^(10\.|192\.168\.|172\.(1[6-9]|2[0-9]|3[01])\.)/ { geoip { source => "remote_addr" target => "geoip" } } } output { elasticsearch { hosts => ["http://elasticsearch-master:9200"] index => "rtf-logs-%{+YYYY.MM.dd}" template_name => "rtf-logs" template_pattern => "rtf-logs-*" template_overwrite => true } # Debug output (remove in production) stdout { codec => rubydebug } } k8s-logs.conf: | input { beats { port => 5045 } } filter { json { source => "message" } mutate { add_field => { "cluster_name" => "rtf-cluster" } } } output { elasticsearch { hosts => ["http://elasticsearch-master:9200"] index => "kubernetes-logs-%{+YYYY.MM.dd}" } } EOF kubectl apply -f logstash-config.yaml # Deploy Logstash cat << EOF > logstash-values.yaml replicas: 3 logstashConfig: logstash.yml: | http.host: 0.0.0.0 xpack.monitoring.elasticsearch.hosts: ["http://elasticsearch-master:9200"] logstashPipeline: rtf-logs.conf: | input { beats { port => 5044 } } filter { # RTF-specific log parsing if [kubernetes][container_name] == "runtime-fabric" { grok { match => { "message" => "%{TIMESTAMP_ISO8601:timestamp} %{LOGLEVEL:level} \[%{DATA:thread}\] %{DATA:logger}: %{GREEDYDATA:log_message}" } } } } output { elasticsearch { hosts => ["http://elasticsearch-master:9200"] index => "rtf-logs-%{+YYYY.MM.dd}" } } resources: requests: cpu: "1000m" memory: "2Gi" limits: cpu: "2000m" memory: "4Gi" service: type: ClusterIP ports: - port: 5044 name: beats - port: 5000 name: tcp - port: 9600 name: http EOF helm install logstash elastic/logstash \ --namespace logging \ --values logstash-values.yaml \ --wait

📊 Step 3: Deploy Kibana

# Create Kibana configuration cat << EOF > kibana-values.yaml elasticsearchHosts: "http://elasticsearch-master:9200" replicas: 2 resources: requests: cpu: "1000m" memory: "2Gi" limits: cpu: "2000m" memory: "4Gi" kibanaConfig: kibana.yml: | server.host: 0.0.0.0 elasticsearch.hosts: ["http://elasticsearch-master:9200"] xpack.security.enabled: true xpack.encryptedSavedObjects.encryptionKey: "something_at_least_32_characters" # Service configuration service: type: LoadBalancer ports: - port: 5601 name: http # Ingress configuration ingress: enabled: true className: "nginx" annotations: nginx.ingress.kubernetes.io/ssl-redirect: "true" cert-manager.io/cluster-issuer: "letsencrypt-prod" hosts: - host: kibana.yourdomain.com paths: - path: / pathType: Prefix tls: - secretName: kibana-tls hosts: - kibana.yourdomain.com EOF # Install Kibana helm install kibana elastic/kibana \ --namespace logging \ --values kibana-values.yaml \ --wait # Create Kibana dashboards and index patterns kubectl exec -n logging deployment/kibana-kibana -- curl -X POST "localhost:5601/api/saved_objects/index-pattern/rtf-logs" \ -H "Content-Type: application/json" \ -H "kbn-xsrf: true" \ -d '{ "attributes": { "title": "rtf-logs-*", "timeFieldName": "@timestamp" } }'

# Configure AlertManager for security incidents cat << EOF > alertmanager-security-config.yaml apiVersion: v1 kind: Secret metadata: name: alertmanager-security-config namespace: monitoring type: Opaque stringData: alertmanager.yml: | global: smtp_smarthost: 'localhost:587' smtp_from: 'alerts@yourdomain.com' route: group_by: ['alertname'] group_wait: 10s group_interval: 10s repeat_interval: 1h receiver: 'security-team' routes: - match: severity: critical receiver: 'critical-alerts' group_wait: 0s receivers: - name: 'security-team' email_configs: - to: 'security@yourdomain.com' subject: 'RTF Security Alert: {{ .GroupLabels.alertname }}' body: | {{ range .Alerts }} Alert: {{ .Annotations.summary }} Description: {{ .Annotations.description }} Instance: {{ .Labels.instance }} Severity: {{ .Labels.severity }} {{ end }} - name: 'critical-alerts' email_configs: - to: 'security@yourdomain.com,oncall@yourdomain.com' subject: 'CRITICAL RTF Security Alert: {{ .GroupLabels.alertname }}' slack_configs: - api_url: 'https://hooks.slack.com/services/YOUR/SLACK/WEBHOOK' channel: '#security-alerts' title: 'RTF Critical Security Alert' text: 'Critical security issue detected in RTF cluster' EOF kubectl apply -f alertmanager-security-config.yaml

📝 Log Management & Retention

📚 Log Categories

  • Application Logs: Mule application events
  • System Logs: RTF infrastructure events
  • Audit Logs: Security and compliance events
  • Performance Logs: Metrics and monitoring data

🗓️ Retention Policies

  • Critical Logs: 2 years retention
  • Application Logs: 90 days retention
  • Debug Logs: 30 days retention
  • Archived Logs: 7 years cold storage
# Log retention policy configuration apiVersion: v1 kind: ConfigMap metadata: name: log-retention-policy namespace: logging data: retention-config.yaml: | policies: critical_logs: retention_days: 730 storage_tier: "hot" backup_enabled: true application_logs: retention_days: 90 storage_tier: "warm" compression_enabled: true debug_logs: retention_days: 30 storage_tier: "hot" auto_delete: true audit_logs: retention_days: 2555 # 7 years storage_tier: "cold" encryption_required: true compliance_mode: true
